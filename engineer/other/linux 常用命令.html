# linux 常用命令

// 切换目录
cd xxx

// 查看当前目录下文件
ls

// 查看当前目录路径
pwd

// 查看当前文件内容
cat xxx

// 修改文件、文件夹名，或移动文件
mv xxx xxx

// 复制文件
cp xxx xxx_copy

// 新建文件
touch xxx

// 新建文件夹
mkdir xxx

// 删除文件
rm xxx

// 删除文件夹及其下所有文件
rm -rf xxx

// 进入文本编辑
i

// 退出文本编辑
:wq
