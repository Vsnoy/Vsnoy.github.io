# 作用域

var a = 1

function fn() {
  console.log('1:' + a) // '1:3'
  a = 2  // fn 执行后才会生效，全局作用域的 a 才会被改为 2
}

a = 3

function bar() {
  console.log('2:' + a)  // '2:2'
}

fn()
bar()
