# 习题

function createIncrement() {
  let count = 0
  function increment() { 
    count++
    console.log(count)
  }

  let message = count
  function log() {
    console.log('message: ', message)
    console.log('count: ', count)
  }
  
  return [increment, log]
}

const [increment, log] = createIncrement()

increment() 
increment()
increment() 

log()

// 输出
1
2
3
message: 0
count: 3
