# 习题

function createIncrement() {
  let count = 0
  function increment() { 
    count++
    console.log(count)
  }

  let message = `Count is ${count}`
  function log() {
    console.log(message)
  }
  
  return [increment, log]
}

const [increment, log] = createIncrement()
increment() 
log()

// 输出
1
2
3
Count is 0
