# this 指向

## 习题一

```
var name = 'violet'
var object = {
  name : 'trauma',
  getName: function() {
    return this.name
  }
}

console.log(object.getName()) 
console.log((object.getName = object.getName)())

// 输出
trauma
violet

// 难点
赋值语句返回值是等号右边的值
```

## 习题二

```
var tag = 'window'

function getTag() {
  console.log(tag)
  console.log(this.tag)
}

const obj1 = (function(){
  const tag = '1-1'
  return {
    tag: '1-2',
    getTag(){
      console.log(tag)
      console.log(this.tag)
    }
  }
})()

const obj2 = (function(){
  const tag = '2-1'
  return {
    tag: '2-2',
    getTag(){
      console.log(tag)
      console.log(this.tag)
    }
  }
})()

getTag()
obj1.getTag()
obj2.getTag()
obj1.getTag = getTag
obj1.getTag()
obj1.getTag = obj2.getTag
obj1.getTag()

// 输出结果
window
1-1
1-2
2-1
2-2
window
1-2
2-1
1-2
```
